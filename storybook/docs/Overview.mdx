export const ComponentCard = ({ name, description, link, story, previewHeight = 154, previewScale = 0.9 }) => {
  const handleCardClick = (e) => {
    const previewArea = e.currentTarget.querySelector('[data-preview-area]');
    if (!previewArea || !previewArea.contains(e.target)) {
      window.top.location.href = link;
    }
  };

  return (
    <div
      style={{
        display: 'flex', flexDirection: 'column', padding: 20, borderRadius: 12,
        background: 'var(--sb-bg-subtle)', border: '1px solid var(--sb-border)',
        transition: 'all .2s ease', cursor: 'pointer'
      }}
      onClick={handleCardClick}
      onMouseOver={(e) => { e.currentTarget.style.borderColor = 'var(--sb-code-color)'; e.currentTarget.style.transform = 'translateY(-2px)'; e.currentTarget.style.boxShadow = '0 4px 12px rgba(0,0,0,.1)'; }}
      onMouseOut={(e) => { e.currentTarget.style.borderColor = 'var(--sb-border)'; e.currentTarget.style.transform = 'translateY(0)'; e.currentTarget.style.boxShadow = 'none'; }}
    >
      <div
        data-preview-area="true"
        style={{ display: 'flex', alignItems: 'center', justifyContent: 'center', height: previewHeight, background: 'var(--sb-bg)', borderRadius: 8, marginBottom: 16, overflow: 'hidden', border: '1px solid var(--sb-border-subtle)' }}
      >
        <div style={{ transform: `scale(${previewScale})` }}><Story of={story} /></div>
      </div>
      <div style={{ fontWeight: 'bold', fontSize: 15, marginBottom: 6 }}>{name}</div>
      <div style={{ fontSize: 12, color: 'var(--sb-text-subtle)', lineHeight: 1.4, marginBottom: 12 }}>{description}</div>
      <div style={{ marginTop: 'auto', paddingTop: 8, fontSize: 11, color: 'var(--sb-code-color)', fontWeight: 'bold' }}>Voir la documentation →</div>
    </div>
  );
};
