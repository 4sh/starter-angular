# Couleurs

Cette page est **générée à partir des tokens** (`src/styles/src/generated/tokens.manifest.json`,
produit par `npm run tokens:build`). Ajoute / retire un token dans `design-tokens/*.json` →
la doc se met à jour automatiquement. Les noms listés sont les **variables CSS générées**
(copiables). Le switch mode/marque n'affecte **que les aperçus et valeurs** (scope local).

export const COLORS = manifest.tokens.filter((t) => t.type === 'color');
export const GROUPED = (() => {
  const g = {};
  for (const t of COLORS) {
    (g[t.collection] = g[t.collection] || {});
    (g[t.collection][t.category] = g[t.collection][t.category] || []).push(t.name);
  }
  for (const col in g) for (const cat in g[col]) g[col][cat].sort((a, b) => a.localeCompare(b, undefined, { numeric: true }));
  return g;
})();
export const TOPS = Object.keys(GROUPED);
export const cap = (s) => s.charAt(0).toUpperCase() + s.slice(1);

export const parseRgb = (str) => {
  const m = String(str).match(/-?[\d.]+/g);
  if (!m || m.length < 3) return null;
  const [r, g, b] = m.map(Number);
  const a = m.length >= 4 ? Number(m[3]) : 1;
  return { r, g, b, a };
};
export const toHex = ({ r, g, b, a }) => {
  const h = (x) => Math.round(x).toString(16).padStart(2, '0');
  let s = `#${h(r)}${h(g)}${h(b)}`;
  if (a != null && a < 1) s += h(Math.round(a * 255));
  return s.toUpperCase();
};
export const toOklch = ({ r, g, b, a }) => {
  const lin = (c) => { c /= 255; return c <= 0.04045 ? c / 12.92 : Math.pow((c + 0.055) / 1.055, 2.4); };
  const R = lin(r), G = lin(g), B = lin(b);
  const l = 0.4122214708 * R + 0.5363325363 * G + 0.0514459929 * B;
  const m = 0.2119034982 * R + 0.6806995451 * G + 0.1073969566 * B;
  const s = 0.0883024619 * R + 0.2817188376 * G + 0.6299787005 * B;
  const l_ = Math.cbrt(l), m_ = Math.cbrt(m), s_ = Math.cbrt(s);
  const L = 0.2104542553 * l_ + 0.7936177850 * m_ - 0.0040720468 * s_;
  const A = 1.9779984951 * l_ - 2.4285922050 * m_ + 0.4505937099 * s_;
  const Bb = 0.0259040371 * l_ + 0.7827717662 * m_ - 0.8086757660 * s_;
  const C = Math.sqrt(A * A + Bb * Bb);
  let H = Math.atan2(Bb, A) * 180 / Math.PI;
  if (H < 0) H += 360;
  const core = `${(L * 100).toFixed(1)}% ${C.toFixed(3)} ${H.toFixed(1)}`;
  return a != null && a < 1 ? `oklch(${core} / ${Math.round(a * 100)}%)` : `oklch(${core})`;
};
export const formats = (rgbStr) => {
  const c = parseRgb(rgbStr);
  if (!c) return { rgb: rgbStr, hex: '—', oklch: '—' };
  return { rgb: rgbStr, hex: toHex(c), oklch: toOklch(c) };
};

export const ColorsExplorer = () => {
  const [top, setTop] = useState(TOPS[0]);
  const cats = Object.keys(GROUPED[top] || {});
  const [sub, setSub] = useState(cats[0]);
  const [theme, setTheme] = useState('light');
  const [brand, setBrand] = useState('brand1');
  const [resolved, setResolved] = useState({});
  const [copied, setCopied] = useState(null);
  const scopeRef = useRef(null);

  useEffect(() => { setSub(Object.keys(GROUPED[top] || {})[0]); }, [top]);

  const names = (GROUPED[top] && GROUPED[top][sub]) || [];

  useEffect(() => {
    const el = scopeRef.current;
    if (!el || !names.length) return;
    const id = requestAnimationFrame(() => {
      const probe = document.createElement('span');
      probe.style.position = 'absolute'; probe.style.visibility = 'hidden';
      el.appendChild(probe);
      const out = {};
      for (const n of names) { probe.style.color = ''; probe.style.color = `var(--${n})`; out[n] = formats(getComputedStyle(probe).color); }
      probe.remove();
      setResolved(out);
    });
    return () => cancelAnimationFrame(id);
  }, [top, sub, theme, brand]);

  const copy = (n) => { navigator.clipboard?.writeText(`var(--${n})`); setCopied(n); setTimeout(() => setCopied(null), 1200); };

  return (
    <div className="tfm-colors">
      <style>{`
        .tfm-colors { --accent: #9747ff; }
        .tfm-seg { display: inline-flex; padding: 4px; gap: 4px; border-radius: 10px; background: rgba(128,128,128,.12); margin-bottom: 16px; }
        .tfm-seg button { border: 0; background: transparent; padding: 7px 18px; border-radius: 7px; font-size: 14px; font-weight: 600; cursor: pointer; color: inherit; transition: all .15s; }
        .tfm-seg button.is-active { background: var(--accent); color: #fff; box-shadow: 0 1px 3px rgba(0,0,0,.2); }
        .tfm-subtabs { display: flex; flex-wrap: wrap; gap: 8px; margin-bottom: 20px; }
        .tfm-subtabs button { border: 1px solid rgba(128,128,128,.28); background: transparent; color: inherit; padding: 6px 14px; border-radius: 999px; font-size: 13px; cursor: pointer; transition: all .15s; }
        .tfm-subtabs button:hover { border-color: var(--accent); color: var(--accent); }
        .tfm-subtabs button.is-active { background: var(--accent); border-color: var(--accent); color: #fff; }
        .tfm-top-row { display: flex; justify-content: space-between; align-items: center; margin-bottom: 16px; flex-wrap: wrap; gap: 16px; }
        .tfm-switches { display: flex; gap: 24px; flex-wrap: wrap; align-items: center; }
        .tfm-switches .grp { display: flex; gap: 6px; align-items: center; }
        .tfm-switches .grp > span { font-size: 12px; font-weight: 700; opacity: .7; }
        .tfm-switches button { border: 1px solid rgba(128,128,128,.28); background: transparent; color: inherit; padding: 5px 12px; border-radius: 7px; font-size: 13px; cursor: pointer; }
        .tfm-switches button.is-active { background: var(--accent); border-color: var(--accent); color: #fff; }
        .tfm-table-wrap { overflow-x: auto; }
        .tfm-table { width: 100%; border-collapse: separate; border-spacing: 0; font-size: 13px; }
        .tfm-val { opacity: .8; font-variant-numeric: tabular-nums; font-family: var(--sb-code-font, ui-monospace, monospace); font-size: 12px; white-space: nowrap; }
        .tfm-table thead th { position: sticky; top: 0; z-index: 1; text-align: left; padding: 10px 14px; border-bottom: 1px solid rgba(128,128,128,.22); font-size: 11.5px; font-weight: 700; letter-spacing: .04em; text-transform: uppercase; opacity: .7; background: var(--sb-bg-subtle, #f8fafc); }
        .tfm-table td { padding: 10px 14px; border-bottom: 1px solid rgba(128,128,128,.12); vertical-align: middle; }
        .tfm-colors-stack { display: flex; flex-direction: column; gap: 2px; }
        .tfm-colors-stack .hex { font-weight: 600; }
        .tfm-colors-stack .small { font-size: 11px; opacity: .7; }
        .tfm-table tbody tr:last-child td { border-bottom: none; }
        .tfm-table tbody tr { transition: background-color .15s ease; }
        .tfm-table tbody tr:nth-child(even) { background-color: rgba(128,128,128,.04); }
        .tfm-table tbody tr:hover { background-color: rgba(151,71,255,.08); }
        .tfm-sw { display: block; width: 40px; height: 26px; border-radius: 6px; border: 1px solid rgba(128,128,128,.3); }
        .tfm-copy { font-size: 12px; padding: 3px 10px; border-radius: 6px; cursor: pointer; border: 1px solid rgba(128,128,128,.3); background: transparent; color: inherit; transition: all .15s; }
        .tfm-copy:hover { border-color: var(--accent); color: var(--accent); }
      `}</style>

      <div className="tfm-top-row">
        <div className="tfm-seg">
          {TOPS.map((t) => (
            <button key={t} className={top === t ? 'is-active' : ''} onClick={() => setTop(t)}>{cap(t)}</button>
          ))}
        </div>

        <div className="tfm-switches">
          <div className="grp"><span>Mode</span>
            {[['light','Light'],['dark','Dark']].map(([v,l]) => <button key={v} className={theme===v?'is-active':''} onClick={() => setTheme(v)}>{l}</button>)}
          </div>
          <div className="grp"><span>Marque</span>
            {[['brand1','Brand 1'],['brand2','Brand 2'],['brand3','Brand 3']].map(([v,l]) => <button key={v} className={brand===v?'is-active':''} onClick={() => setBrand(v)}>{l}</button>)}

      </div>

      <div className="tfm-subtabs">
        {cats.map((c) => (
          <button key={c} className={sub === c ? 'is-active' : ''} onClick={() => setSub(c)}>{cap(c)}</button>
        ))}
      </div>

      <div ref={scopeRef} data-theme={theme} data-brand={brand !== 'brand1' ? brand : undefined}>
        <div className="tfm-table-wrap">
        <table className="tfm-table">
          <thead>
            <tr><th style={{ width: 56 }}></th><th>Token (variable CSS)</th><th>Valeurs</th><th style={{ width: 90 }}></th></tr>
          </thead>
          <tbody>
            {names.map((n) => (
              <tr key={n}>
                <td><span className="tfm-sw" style={{ background: `var(--${n})` }} /></td>
                <td><code>--{n}</code></td>
                <td>
                  {resolved[n] ? (
                    <div className="tfm-colors-stack">
                      <span className="tfm-val hex">{resolved[n].hex}</span>
                      <span className="tfm-val small">{resolved[n].rgb}</span>
                      <span className="tfm-val small">{resolved[n].oklch}</span>
                    </div>
                  ) : '…'}
                </td>
                <td><button className="tfm-copy" onClick={() => copy(n)}>{copied === n ? '✓ copié' : 'copier'}</button></td>
              </tr>
            ))}
          </tbody>
        </table>
        </div>
        <p style={{ fontSize: 12, opacity: .6, marginTop: 12 }}>{names.length} token(s) · catégorie <code>{sub}</code></p>
      </div>
    </div>
  );
};
